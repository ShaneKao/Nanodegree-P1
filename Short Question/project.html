import numpy as np
raw_data=pd.read_csv('turnstile_weather_v2.csv')

without_rain_mean=np.mean(data_test[data_test['rain']==0]['ENTRIESn_hourly'])
print('The mean ENTRIESn_hourly with rain is '+str(with_rain_mean))
print('The mean ENTRIESn_hourly without rain is '+str(without_rain_mean))
x=data_test[data_test['rain']==1]['ENTRIESn_hourly']
y=data_test[data_test['rain']==0]['ENTRIESn_hourly']
u, prob=scipy.stats.mannwhitneyu(x,y)
m_u = len(x)*len(y)/2
sigma_u = np.sqrt(len(x)*len(y)*(len(x)+len(y)+1)/12)
z = (u - m_u)/sigma_u
pval = 2*scipy.stats.norm.cdf(z)
print('The p-value='+str(pval))

import numpy as np
import pandas as pd
import scipy
import statsmodels.api as sm
raw_data=pd.read_csv('turnstile_weather_v2.csv')
dummy_units=pd.get_dummies(raw_data['UNIT'], prefix='unit')
features=raw_data[['rain']].join(dummy_units)
values=raw_data[['ENTRIESn_hourly']]
m = len(values)
features['ones'] = np.ones(m)
features_array=np.array(features)
values_array = np.array(values).flatten()
X=features_array
Y=values_array
model = sm.OLS(Y,X).fit()
print('The coefficient of rain is '+str(model.params[0]))

The coefficient of rain is 173.959768089
